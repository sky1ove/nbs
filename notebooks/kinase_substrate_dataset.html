def plot_corr(
              x,
              y,
              xlabel=None,# x axis label
              ylabel=None,# y axis label
              order=1, # polynomial level, if straight, order=1
              method = 'pearson', # can be spearman
             ):
    "Given a dataframe and the name of two columns, plot the two columns' correlation"
    
    if method == 'spearman':
        correlation, pvalue = spearmanr(x, y)
    elif method == 'pearson':
        correlation, pvalue = pearsonr(x, y)
    else:
        raise ValueError("Invalid correlation method. Choose 'spearman' or 'pearson'.")
        
    sns.regplot(
                x=x,
                y=y,
                order=order,
                line_kws={'color': 'gray'}
           )
    
    if xlabel is not None:
        plt.xlabel(xlabel)
        
    if ylabel is not None:
        plt.ylabel(ylabel)
    
    correlation_text = f'Spearman: {correlation:.2f}' if method == 'spearman' else f'Pearson: {correlation:.2f}'

    plt.text(x=0.8, y=0.1, s=correlation_text, transform=plt.gca().transAxes, ha='center', va='center')
    
    plt.show()

Get heatmap of specific kinase

def get_freq(df3, k):
    df_k = df3.query(f'KINASE=="{k}"').reset_index()
    
    #Count frequency for each amino acid at each position
    melted_k = df_k.melt(id_vars=['KINASE', 'SITE_+/-7_AA'], 
                    value_vars=[i for i in range(-7, 8)],
                    var_name='Position', 
                    value_name='aa')
    
    # Group by Position and Amino Acid and count occurrences
    grouped = melted_k.groupby(['Position', 'aa']).size().reset_index(name='Count')
    
    # get pivot table
    pivot_k = grouped.pivot(index='aa', columns='Position', values='Count').fillna(0)
    
    # Get frequency by dividing the sum of each column
    freq_k = pivot_k/pivot_k.sum()

    
    # data from the kinase-substrate dataset, and format is Lew's paper's format
    p_k_paper = freq_k.reindex(index=aa_order_paper)[position_paper]

    # full pivot data from kinase-substrate dataset
    p_k = freq_k.reindex(index=aa_order)[position]

    
    return p_k_paper, p_k
def plot_raw_ks(k):
    
    p_k_paper, p_k = get_freq(df3,k)
    raw_unavailable = []
    
    k=k.upper() # kinase in raw PSPA data is in upper case
    
    # PSPA data from Lew's paper
    try:     
        p_raw = get_one_kinase(raw,k).T
        p_raw = p_raw.reindex(index=aa_order_paper)
        x = p_raw.unstack().values
        y = p_k_paper.unstack().values
        
        # draw raw vs. kinase-substrate correlation
        plt.title(f'{k}')
        plot_corr(x,y, xlabel='PSPA',ylabel='kinase-substrate dataset')
        
        # draw raw PSPA data
        plt.figure(figsize=(6, 10))
        sns.heatmap(p_raw, cmap='binary', annot=False)
        plt.title(f'{k} from PSPA data')
        plt.show()
        
    except:
        print(f'{k} is not available in PSPA data')
        raw_unavailable.append(k)

    # draw kinase-substrate heatmap using paper's format
    plt.figure(figsize=(6, 10))
    sns.heatmap(p_k_paper, cmap='binary', annot=False)
    plt.title(f'{k} from kinase substrate dataset')
    plt.show()

    # draw full heatmap
    plt.figure(figsize=(9, 11))
    sns.heatmap(p_k, cmap='binary', annot=False)
    plt.title(f'{k} from kinase substrate dataset (full)')
    plt.show()
    
    return raw_unavailable
